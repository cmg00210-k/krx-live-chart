  \vspace{4pt}%
  \noindent%
  \begin{tabularx}{\textwidth}{@{}XXXXX@{}}
    \stagenavbox{1}{문제 정의}{stageOneMarker}{#1} &
    \stagenavbox{2}{데이터 기반}{stageTwoMarker}{#1} &
    \stagenavbox{3}{분석 엔진}{stageThreeMarker}{#1} &
    \stagenavbox{4}{검증·통제}{stageFourMarker}{#1} &
    \stagenavbox{5}{로드맵}{stageFiveMarker}{#1}
  \end{tabularx}%
  \vspace{6pt}\par%
}

\newcommand{\stagenavbox}[4]{%
  % #1=number, #2=label, #3=color, #4=current stage number
  \ifnum#1=#4\relax%
    \colorbox{#3}{%
      \parbox[c][10mm]{\dimexpr\linewidth-2\fboxsep\relax}{\centering\color{white}%
        \fontsize{7.5pt}{9pt}\selectfont\sffamily\bfseries Stage #1\\[-1pt]%
        \fontsize{6.5pt}{8pt}\selectfont #2%
      }%
    }%
  \else%
    \colorbox{tableodd}{%
      \parbox[c][10mm]{\dimexpr\linewidth-2\fboxsep\relax}{\centering\color{muted}%
        \fontsize{7.5pt}{9pt}\selectfont\sffamily Stage #1\\[-1pt]%
        \fontsize{6.5pt}{8pt}\selectfont #2%
      }%
    }%
  \fi%
}
